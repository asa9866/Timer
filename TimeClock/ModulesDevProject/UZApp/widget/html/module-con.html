moduelDemo.showAlert 显示提示框

var param = {
    msg:"Hello App!"
};
var demo = api.require('moduleDemo');
demo.showAlert(param, callBack);

function callBack(ret, err){
    var msg;
    if (ret.index == 1){
        msg = "点击了第一个按钮";
    } else {
        msg = "点击了第二个按钮";
    }
    api.toast({
        msg:msg
    });
}
